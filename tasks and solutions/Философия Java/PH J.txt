class Main {

    public static void main(String[] args){
        Test test1 = new Test(1, 2, 3, 4);
        Test test2 = new Test(5, 6, 7, 8);
        Test test3 = new Test(9,10,11,12);
        System.out.println(test1.i0);
        System.out.println(test2.i0);
        System.out.println(test3.i0);
    }


    public static class Test{
        static int i0;
        int i;
        int i2;
        int i3;
        Test(int i, int i0, int i2, int i3){
            this.i=i;
            this.i0=i0;
            this.i2=i2;
            this.i3=i3;
        }
    }
}
